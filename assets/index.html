try { var ss = SpreadsheetApp.getActiveSpreadsheet(); // 2. 寫入登入紀錄 (登入時間 | IP | 姓名 | 權限等級) var logSheet = ss.getSheetByName("使用者登入紀錄"); if (logSheet) { logSheet.appendRow([new Date(), userIp, currentUser.name, currentUser.level]); } // 3. 讀取資料 var sheet = ss.getSheetByName(currentUser.targetSheet); // 💡 這裡增加錯誤判斷：如果找不到分頁，回傳具體錯誤文字 if (!sheet) { return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": "權限配置錯誤：找不到名為 [" + currentUser.targetSheet + "] 的工作表" })).setMimeType(ContentService.MimeType.JSON); } var fullData = sheet.getDataRange().getValues(); if (fullData.length < 1) { throw new Error("工作表內容為空"); } var headers = fullData[0]; var rows = fullData.slice(1); var jsonData = rows.map(function(row) { var obj = {}; headers.forEach(function(header, i) { obj[header.toString().trim()] = row[i]; }); return obj; }); return ContentService.createTextOutput(JSON.stringify({ "status": "success", "userName": currentUser.name, "userLevel": currentUser.level, "headers": headers, "data": jsonData })).setMimeType(ContentService.MimeType.JSON); } catch (err) { // 💡 這裡會把具體的錯誤訊息傳回給網頁，讓你知道為什麼「連線失敗」 return ContentService.createTextOutput(JSON.stringify({ "status": "error", "message": "系統錯誤: " + err.message })).setMimeType(ContentService.MimeType.JSON); }
